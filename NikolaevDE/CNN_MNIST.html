def relu(x):
    xx = x.copy()
    
    for i in range(len(xx)):
        for j in range(len(xx[i])):
            xx[i][j] = max(0, xx[i][j])
            
    return xx

def derivative_relu(x):
    xx = x.copy()
    
    for i in range(len(xx)):
        for j in range(len(xx[i])):            
            xx[i][j] = float(xx[i][j] > 0)

    return xx

def softmax(u):
    shift = np.max(u, axis=u.ndim - 1, keepdims=True)
    u = np.exp(u - shift)
    denominator = np.sum(u, axis=u.ndim - 1, keepdims=True)

    return u / denominator

class Network():
    def __init__(self, input_neurons_size, hidden_neurons_size, out_neurons_size, learning_rate):
        self.input_neurons_size  = input_neurons_size
        self.hidden_neurons_size = hidden_neurons_size
        self.out_neurons_size    = out_neurons_size
        self.learning_rate       = learning_rate
        
        mean = 0.0
        stdw = pow(hidden_neurons_size, -0.5)
        stdb = pow(hidden_neurons_size, -0.5)
        
        #  
        self.w0 = np.random.normal(0.0, stdw, size=(input_neurons_size, hidden_neurons_size))
        self.b0 = np.random.normal(0.0, stdb, size=(hidden_neurons_size))
        
        #  
        self.w1 = np.random.normal(0.0, stdw, size=(hidden_neurons_size, out_neurons_size))
        self.b1 = np.random.normal(0.0, stdb, size=(out_neurons_size))

    #  
    def forward(self, x):
        x = np.matmul(x, self.w0) + self.b0
        
        self.w0_1 = x.copy()
        
        x = relu(x)
        
        self.w1_1 = x.copy()

        x = np.matmul(x, self.w1) + self.b1
        x = softmax(x)
        
        return x
    
    #  
    def backward(self, x, y, y_predicted):
        e1 = y_predicted - y
        e0 = np.matmul(e1, self.w1.T) * derivative_relu(self.w0_1)

        delta = 1.0 / len(x)

        self.w0 = self.w0 - self.learning_rate * delta * np.matmul(x.T, e0)
        self.b0 = self.b0 - self.learning_rate * delta * np.sum(e0, axis=0)
        
        delta = 0.925 / len(x)

        self.w1 = self.w1 - self.learning_rate * delta * np.matmul(self.w1_1.T, e1)
        self.b1 = self.b1 - self.learning_rate * delta * np.sum(e1, axis=0)
        
    def training_iteration(self, x, y):
        y_predicted = self.forward(x)
        
        self.backward(x, y, y_predicted)
        
    def validate(self, x, y):
        y_predicted = self.forward(x)
        
        loss = compute_cross_entropy(y, y_predicted)
        accuracy = compute_accuracy(y, y_predicted)
        
        return loss, accuracy

def train(network, train_x, train_y, epochs=20, batch_size=8):
    
    table_data = []
    
    start = datetime.now()

    for epoch_index in range(epochs):
        ########################################################
        
        epoch_start = datetime.now()
        
        for offset in range(0, len(train_x), batch_size):

            batch_x = train_x[offset:offset + batch_size]
            batch_y = train_y[offset:offset + batch_size]

            network.training_iteration(batch_x, batch_y)
        
        epoch_end = datetime.now()
        
        ########################################################
        clear_output(wait=True)
        
        loss, accuracy = network.validate(train_x, train_y)

        table_data.append([
            epoch_index,
            '{:1.8f}'.format(loss), 
            '{:1.8f}'.format(accuracy),
            '{:1.6f} '.format((epoch_end - epoch_start).total_seconds())
        ])

        print(tabulate(table_data, headers=["", "", ""," "]))

    end = datetime.now()
    time = (end - start).total_seconds()

    print('\n  {} '.format(time))

class Torch_NN(nn.Module):
    def __init__(self, input_neurons_size, hidden_neurons_size, out_neurons_size, learning_rate):
        super().__init__()
        
        self.l1   = nn.Linear(input_neurons_size, hidden_neurons_size)
        self.l2   = nn.Linear(hidden_neurons_size, out_neurons_size)
        self.relu = nn.ReLU()
        self.loss = nn.CrossEntropyLoss()
        
        self.optimizer = torch.optim.SGD(self.parameters(), lr = learning_rate) 
        
    def forward(self, x):
        x = self.l1(x)
        x = self.relu(x)
        x = self.l2(x)
        x = torch.nn.functional.log_softmax(x)
        
        return x

    def backward(self, x, y, y_predicted):
        loss = self.loss(y_predicted, y)
        self.optimizer.zero_grad() 
        loss.backward() 
        self.optimizer.step()

    def training_iteration(self, x, y):
        x = Variable(torch.tensor(x))
        y = Variable(torch.tensor(y))
        
        y_predicted = self(x)
        
        self.backward(x, y, y_predicted)
        
    def validate(self, x, y):
        x = Variable(torch.tensor(x))
        y = Variable(torch.tensor(y))
        
        y_predicted = self(x)
        
        loss     = self.loss(y_predicted, y).data
        accuracy = compute_accuracy(y.detach().numpy(), y_predicted.detach().numpy())
        
        return loss, accuracy
